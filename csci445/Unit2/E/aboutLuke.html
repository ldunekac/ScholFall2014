For fun I program microcontrollers. My current project is making my own Game Boy ® with the TI Connected Launchpad.

If your not familiar with microcontroller code. Here is the blinking light example for the TI Connected Launchpad.

#include <stdint.h>
#include "inc/tm4c1294ncpdt.h"

int main(void)
{
  volatile uint32_t ui32Loop;

  SYSCTL_RCGCGPIO_R = SYSCTL_RCGCGPIO_R12;

  ui32Loop = SYSCTL_RCGCGPIO_R;

  GPIO_PORTN_DIR_R = 0x01;
  GPIO_PORTN_DEN_R = 0x01;

  while(1)
  {
   GPIO_PORTN_DATA_R |= 0x01;
   for(ui32Loop = 0; ui32Loop < 200000; ui32Loop++);
   GPIO_PORTN_DATA_R &= ~(0x01);
   for(ui32Loop = 0; ui32Loop < 200000; ui32Loop++);
  }
}
